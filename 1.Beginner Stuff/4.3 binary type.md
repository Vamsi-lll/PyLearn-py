# =============================================
# Binary Type - Data Types in Python
# =============================================

# Binary Types are used to store data in binary form (0s and 1s).
# These data types are mostly used when working with:
# - Files
# - Images
# - Network data
# - Low-level data handling

# In Python, Binary Types are:
# 1. bytes
# 2. bytearray


# =============================================
# bytes type
# =============================================

# bytes is a binary data type used to store a sequence of bytes.
# It belongs to the class <class 'bytes'>
# bytes data is immutable (cannot be changed after creation)
# Values are written using a b prefix

# Example:
data_bytes = b"hello"

# Each character is stored as a byte
# bytes values range from 0 to 255


# =============================================
# bytearray type
# =============================================

# bytearray is similar to bytes, but it is mutable.
# It belongs to the class <class 'bytearray'>
# We can modify the data after creation

# Example:
data_bytearray = bytearray(b"hello")

# Since bytearray is mutable, we can change its values
data_bytearray[0] = 72  # ASCII value of 'H'
